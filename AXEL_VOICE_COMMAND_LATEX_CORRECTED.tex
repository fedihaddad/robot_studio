\section{Voice Command Architecture}

The AXEL humanoid robot uses a multi-stage voice command pipeline:

\begin{enumerate}
    \item \textbf{Gemini LLM}: Processes natural language voice input
    \item \textbf{IntentAction Publisher}: Converts LLM intent to structured ROS2 message
    \item \textbf{Action Manager Node}: Priority-based queue with conflict resolution
    \item \textbf{Specialized Dispatchers}: Routes to HeadCommand, HandCommand, or ArmCommand
    \item \textbf{ESP32 Bridge}: Translates to hardware-specific servo commands
\end{enumerate}

\end{lstlisting}

\section{Complete Voice Command Processing Algorithm}

\begin{algorithm}
\caption{AXEL Voice Command Processing Pipeline}
\begin{algorithmic}[1]

\Procedure{ProcessVoiceCommand}{voiceInput}
    \State $\text{intent} \gets \text{GeminiLLM.processNaturalLanguage}(\text{voiceInput})$
    \State $\text{functionName} \gets \text{intent.getFunction}()$ \Comment{e.g., "ouvrir\_bouche"}
    \State $\text{priority} \gets \text{intent.getPriority}()$ \Comment{$0..3$}
    
    \State \textbf{create} $\text{IntentAction}$ message
    \State $\text{msg.source} \gets$ "gemini"
    \State $\text{msg.action\_type} \gets \text{mapActionType}(\text{functionName})$
    \State $\text{msg.function\_name} \gets \text{functionName}$
    \State $\text{msg.priority} \gets \text{priority}$
    \State $\text{msg.parameters\_json} \gets$ additionalParams
    
    \State \textbf{publish} $\text{msg}$ to `/axel/intent_action`
    
\EndProcedure

\newline

\Procedure{ActionManager\_OnIntentAction}{msg}
    \State $\text{bodyGroups} \gets \text{resolveBodyGroups}(\text{msg.action\_type})$
    
    \If{$\text{noBodyGroupsResolved}(\text{bodyGroups})$}
        \State \textbf{publish} `ActionFeedback` with status="failed"
        \State \textbf{return}
    \EndIf
    
    \State \textbf{create} $\text{QueuedAction}$ entry
    \State $\text{action.priority} \gets \text{msg.priority}$
    \State $\text{action.body\_groups} \gets \text{bodyGroups}$
    \State $\text{action.timeout} \gets \text{msg.timeout} \lor \text{DEFAULT\_TIMEOUT}$
    
    \State \textbf{enqueue}($\text{action}$) \Comment{Per body group}
    \State \textbf{checkPreemption}($\text{action}$) \Comment{Cancel lower-priority running actions}
    
\EndProcedure

\newline

\Procedure{DispatchTick}{} \Comment{Called at 20 Hz}
    \For{each $\text{bodyGroup}$ in queues}
        \If{$\text{activeAction}[\text{bodyGroup}] \neq \text{null}$} \textbf{and} bodyGroup not in NON\_BLOCKING
            \State \textbf{continue} \Comment{Wait for action to complete}
        \EndIf
        
        \If{queue[$\text{bodyGroup}$] is empty}
            \State \textbf{continue}
        \EndIf
        
        \State $\text{action} \gets$ queue[$\text{bodyGroup}$].pop() \Comment{Highest priority}
        
        \If{$\text{action.timeout} < \text{now()}$}
            \State \textbf{publish} `ActionFeedback` with status="timeout"
            \State \textbf{continue}
        \EndIf
        
        \If{$\neg \text{allBodyGroupsAvailable}(\text{action})$}
            \State queue[$\text{bodyGroup}$].insert(0, action) \Comment{Requeue}
            \State \textbf{continue}
        \EndIf
        
        \State $\text{action.status} \gets$ "running"
        \State $\text{activeAction}[\text{bodyGroup}] \gets \text{action}$
        
        \State \textbf{dispatchAction}($\text{action}$) \Comment{Send to executor}
        
    \EndFor
\EndProcedure

\newline

\Procedure{DispatchAction}{action}
    \State $\text{esp32Cmd} \gets$ GEMINI\_TO\_ESP32[$\text{action.function\_name}$]
    
    \Switch{$\text{action.action\_type}$}
        \Case{"head", "eyes", "jaw", "expression"}
            \State $\text{msg} \gets$ createHeadCommand()
            \State $\text{msg.expression} \gets \text{esp32Cmd}$
            \State $\text{msg.command\_type} \gets$ "expression" \textbf{or} "position"
            \State $\text{msg.priority} \gets \text{action.priority}$
            \State \textbf{publish} $\text{msg}$ to `/axel/head/cmd`
        \EndCase
        
        \Case{"hand"}
            \State $\text{msg} \gets$ createHandCommand()
            \State $\text{msg.side} \gets$ parameters["side"] \Comment{"left" | "right" | "both"}
            \State $\text{msg.command\_type} \gets$ parameters["command\_type"]
            \State $\{\text{thumb}, \text{index}, \ldots, \text{pinky}\} \gets$ parameters
            \State \textbf{publish} $\text{msg}$ to `/axel/hands/cmd`
        \EndCase
        
        \Case{"arm"}
            \State $\text{msg} \gets$ createArmCommand()
            \State $\text{msg.side} \gets$ parameters["side"]
            \State $\text{msg.command\_type} \gets$ "position" \textbf{or} "trajectory" \textbf{or} "moveit"
            \State $\{\text{shoulder\_lift}, \text{elbow\_flex}, \ldots\} \gets$ parameters
            \State \textbf{publish} $\text{msg}$ to `/axel/arms/cmd`
        \EndCase
        
        \Case{"vision", "speech", "system"}
            \State \textbf{completeAction}(action, "succeeded") \Comment{Non-blocking}
        \EndCase
        
    \EndSwitch
    
    \State \textbf{publish} `ActionFeedback` with status="running"
\EndProcedure

\end{algorithmic}
\end{algorithm}

\section{Complete Python Implementation Example}

\begin{lstlisting}
# Simplified example: Processing voice -> Robot action
import rclpy
from rclpy.node import Node
from axel_msgs.msg import IntentAction, ActionFeedback, HeadCommand

class VoiceCommandProcessor(Node):
    def __init__(self):
        super().__init__("voice_command_processor")
        
        self.intent_pub = self.create_publisher(
            IntentAction, "/axel/intent_action", 10
        )
        self.feedback_sub = self.create_subscription(
            ActionFeedback, "/axel/action_feedback",
            self.feedback_callback, 10
        )
    
    def voice_to_intent(self, voice_input: str, priority: int = 1):
        """Convert Gemini LLM output to IntentAction."""
        
        # Map voice command to system function
        command_map = {
            "ouvre la bouche": ("ouvrir_bouche", "expression"),
            "ferme la bouche": ("fermer_bouche", "expression"),
            "regarde à gauche": ("regarder_gauche", "eyes"),
            "fais une expression joyeuse": ("expression_joie", "expression"),
            "lève le bras droit": ("lift_arm", "arm"),
        }
        
        func_name, action_type = command_map.get(
            voice_input.lower(), ("unknown", "system")
        )
        
        # Create IntentAction
        msg = IntentAction()
        msg.stamp = self.get_clock().now().to_msg()
        msg.action_id = f"voice_{int(time.time() * 1000)}"
        msg.source = "gemini"
        msg.action_type = action_type
        msg.function_name = func_name
        msg.priority = priority  # Priority can be elevated by user/system
        msg.parameters_json = '{"side": "right"}'  # For arm/hand commands
        
        # Publish to Action Manager
        self.intent_pub.publish(msg)
        self.get_logger().info(
            f"📤 Sent: {func_name} (priority={priority})"
        )
    
    def feedback_callback(self, msg: ActionFeedback):
        """Monitor action execution."""
        self.get_logger().info(
            f"📨 Feedback [{msg.action_id}]: {msg.status} "
            f"(progress={msg.progress:.0%})"
        )
        
        if msg.status == "failed":
            self.get_logger().error(
                f"❌ Action failed: {msg.error_type} - {msg.error_detail}"
            )

# Usage
if __name__ == "__main__":
    rclpy.init()
    node = VoiceCommandProcessor()
    
    # Examples
    node.voice_to_intent("ouvre la bouche", priority=1)
    node.voice_to_intent("fais une expression joyeuse", priority=1)
    node.voice_to_intent("lève le bras droit", priority=2)
    
    rclpy.spin(node)
\end{lstlisting}

\section{Conflict Resolution & Preemption}

When multiple commands target overlapping body groups, the Action Manager follows:

\begin{enumerate}
    \item \textbf{Priority Check}: Higher priority action preempts lower priority
    \item \textbf{Body Group Lock}: Non-blocking groups (vision, speech, system) execute immediately
    \item \textbf{Queue Management}: Max 5 actions per blocking body group; lowest priority evicted if queue full
    \item \textbf{Timeout Handling}: Actions expire after default 30 seconds (configurable)
    \item \textbf{MoveIt Safety}: IntentActions blocked when MoveIt trajectory execution is active
\end{enumerate}
